## Iterators and for...of

Iterators provide a way to access element of a collection one at a time. The for..of loop is a simple and
concise way to iterate over iterable object like arrays , strings , maps , sets and more.

```js
class Countdown {
  constructor(start) {
    this.start = start;
  }

  [Symbol.iterator]() {
    let current = this.start;

    return {
      next() {
        if (current >= 0) {
          return {value: current--; done : false}
        } else {
          return {value : undefined , done : true}
        }
      }
    }
  }
}

const countdown = new Countdown(5);

for (const number of countdown) {
  console.log(number);
}
```

## Generators

## Promises Recap

provide a way to handle asynchronous operations. They represent a value that may be available now , or in the future or never.

```js
function doInSeconds(seconds, callback) {
  setTimeout(callback, seconds * 1000);
}

console.log("Setting the alert in 2 seconds");
doInSeconds(2, () => alert("2 second!"));
```

we can update that into promise like below

```js
function doInSeconds(seconds, callback) {
  return new Promise((resolve, reject) => {
    setTimeout(() => resolve("Success"), seconds * 1000);
  });
}

console.log("Setting the alert in 2 seconds");

doInSeconds(2)
  .then((value) => alert(value))
  .catch((error) => alert(error));
```

and this another example of promise

```js
function giveMeHello() {
  return new Promise((resolve, reject) => {
    resolve("Hello");
  });
}

var promise = giveMeHello();
promise.then((message) => alert(message));
```

## Async/Await

syntax provides a way to work with asynchronous code in more
synchronous fashion, making it easier to read and write.

```js
function doInSeconds(seconds, callback) {
  return new Promise((resolve, reject) => {
    setTimeout(() => resolve("Success"), seconds * 1000);
  });
}

console.log("Setting the alert in 2 seconds");

async function start() {
  try {
    var value = await doInSeconds(2);
    alert(value);
  } catch (e) {}
}

start();
```

## Proper Tails Calls

optimize tail recursion by reusing stack frames for tail calls, reducing the
overhead of recursive calls.

```js
// This an example of recursive function without PTC
function factorial(n) {
  if (n === 0) {
    return 1;
  } else {
    return n * factorial(n - 1);
  }
}

console.log(factorial(5));

// This is a recursive function with PTC
function factorialPTC(n, acc = 1) {
  if (n === 0) {
    return acc;
  } else {
    return factorialPTC(n - 1, n * acc);
  }
}

console.log(factorialPTC(5));
```

## Proxies and Reflect API

provides a way to intercept and customize operation on object ues Proxies,
offering a set of static methods to manipulate object

```js
const target = { name: "Joseph" };
const handler = {
  get: (obj, prop) => {
    console.log(`Trying to get "${prop}" from "${JSON.stringify(obj)}"`);
    return Reflect.get(obj, prop) || "default";
  },
};

const proxy = new Proxy(target, handler);
console.log(`The name is ${proxy.name}`);
```

## Error cause

The Error cause property allows attaching a cause to an Error object, providing additional context for the error.

```js
try {
  throw new Error("Something went wrong", {
    cause: new Error("Original error"),
  });
} catch (error) {
  console.log(error.cause);
}
```

or something like below

```js
class Product {
  name;
  constructor(name , price) {
    if (name?.length<2) {
      throw new Error("Product can't be created.",
        {cause : new Error("Name is invalid")}
      );

    }
    this.name = name;
    try {
      this.price = price *1.30  
    }catch(e) {
        throw new Error("Product can't be created",
          {case : e }
        )
      }
  }
}
```
